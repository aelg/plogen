********** Mapped Logic **********
FDCPE_cr: FDCPE port map (cr,cr_D,clk,'0','0');
     cr_D <= ((reset AND cr)
      OR (NOT reset AND qc_0 AND qc(1) AND qc(2) AND qc(3)));
FDCPE_qc1: FDCPE port map (qc(1),qc_D(1),clk,reset,'0');
     qc_D(1) <= ((NOT strobe_pulse AND qc_0 AND NOT qc(1))
      OR (NOT strobe_pulse AND NOT qc_0 AND qc(1)));
FTCPE_qc2: FTCPE port map (qc(2),qc_T(2),clk,reset,'0');
     qc_T(2) <= ((strobe_pulse AND qc(2))
      OR (NOT strobe_pulse AND qc_0 AND qc(1)));
FTCPE_qc3: FTCPE port map (qc(3),qc_T(3),clk,reset,'0');
     qc_T(3) <= ((strobe_pulse AND qc(3))
      OR (NOT strobe_pulse AND qc_0 AND qc(1) AND qc(2)));
FDCPE_qc_0: FDCPE port map (qc_0,qc_0_D,clk,reset,'0');
     qc_0_D <= (NOT strobe_pulse AND NOT qc_0);
FDCPE_qr0: FDCPE port map (qr(0),qr_D(0),clk,reset,'0');
     qr_D(0) <= ((NOT strobe_pulse AND NOT qr(0) AND NOT cr)
      OR (NOT strobe_pulse AND NOT qr(1) AND cr));
FDCPE_qr1: FDCPE port map (qr(1),qr_D(1),clk,reset,'0');
     qr_D(1) <= ((strobe_pulse AND d(1))
      OR (NOT strobe_pulse AND qr(1) AND NOT cr)
      OR (NOT strobe_pulse AND qr(2) AND cr));
FDCPE_qr2: FDCPE port map (qr(2),qr_D(2),clk,reset,'0');
     qr_D(2) <= ((strobe_pulse AND d(2))
      OR (NOT strobe_pulse AND qr(2) AND NOT cr)
      OR (NOT strobe_pulse AND qr(3) AND cr));
FDCPE_qr3: FDCPE port map (qr(3),qr_D(3),clk,reset,'0');
     qr_D(3) <= ((strobe_pulse AND d(3))
      OR (NOT strobe_pulse AND qr(3) AND NOT cr)
      OR (NOT strobe_pulse AND qr(4) AND cr));
FDCPE_qr4: FDCPE port map (qr(4),qr_D(4),clk,reset,'0');
     qr_D(4) <= ((strobe_pulse AND d(4))
      OR (NOT strobe_pulse AND qr(4) AND NOT cr));
FDCPE_qs_FSM_FFd1: FDCPE port map (qs_FSM_FFd1,qs_FSM_FFd1_D,clk,reset,'0');
     qs_FSM_FFd1_D <= ((qs_FSM_FFd2)
      OR (qs_FSM_FFd1 AND strobe));
FDCPE_qs_FSM_FFd2: FDCPE port map (qs_FSM_FFd2,qs_FSM_FFd2_D,clk,reset,'0');
     qs_FSM_FFd2_D <= (NOT qs_FSM_FFd1 AND strobe AND NOT qs_FSM_FFd2);
FDCPE_strobe_pulse: FDCPE port map (strobe_pulse,strobe_pulse_D,clk,reset,'0');
     strobe_pulse_D <= ((strobe_pulse AND NOT qs_FSM_FFd2)
      OR (NOT qs_FSM_FFd1 AND strobe AND NOT qs_FSM_FFd2));
FDCPE_u: FDCPE port map (u,u_D,clk,'0','0');
     u_D <= ((reset AND u.PIN)
      OR (NOT reset AND qr(0)));
